47 
48  Vector3f cameraCenter_pc = -1.0f * (pose_pointCloud->GetR().t() * pose_pointCloud->GetT());
49  Vector3f cameraCenter_live = -1.0f * (pose_d->GetR().t() * pose_d->GetT());
